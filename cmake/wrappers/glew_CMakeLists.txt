# GLEW CMake wrapper
cmake_minimum_required(VERSION 3.18)
project(glew C)

# GLEW source file
add_library(libglew_static STATIC
    src/glew.c
)

# Include directories
target_include_directories(libglew_static PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}/include
)

# Platform-specific definitions
if(APPLE)
    target_compile_definitions(libglew_static PUBLIC
        GLEW_STATIC
    )
elseif(UNIX)
    target_compile_definitions(libglew_static PUBLIC
        GLEW_STATIC
    )
    target_link_libraries(libglew_static PUBLIC
        ${CMAKE_DL_LIBS}
    )
elseif(WIN32)
    target_compile_definitions(libglew_static PUBLIC
        GLEW_STATIC
    )
endif()

# Find and link OpenGL
find_package(OpenGL REQUIRED)
target_link_libraries(libglew_static PUBLIC OpenGL::GL)
