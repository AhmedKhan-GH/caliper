# ImPlot CMake wrapper
cmake_minimum_required(VERSION 3.18)
project(implot CXX)

# ImPlot source files
set(IMPLOT_SOURCES
    implot.cpp
    implot_items.cpp
    implot_demo.cpp
)

# ImPlot headers
set(IMPLOT_HEADERS
    implot.h
    implot_internal.h
)

# Create ImPlot library
add_library(implot STATIC
    ${IMPLOT_SOURCES}
    ${IMPLOT_HEADERS}
)

# Include directories
target_include_directories(implot PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}
)

# Link ImGui (ImPlot depends on ImGui)
target_link_libraries(implot PUBLIC imgui)

# Set C++ standard
set_target_properties(implot PROPERTIES
    CXX_STANDARD 17
    CXX_STANDARD_REQUIRED ON
)
