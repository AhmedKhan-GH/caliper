# ImPlot3D CMake wrapper
cmake_minimum_required(VERSION 3.18)
project(implot3d CXX)

# ImPlot3D source files
set(IMPLOT3D_SOURCES
    implot3d.cpp
    implot3d_items.cpp
    implot3d_meshes.cpp
    implot3d_demo.cpp
)

# ImPlot3D headers
set(IMPLOT3D_HEADERS
    implot3d.h
    implot3d_internal.h
)

# Create ImPlot3D library
add_library(implot3d STATIC
    ${IMPLOT3D_SOURCES}
    ${IMPLOT3D_HEADERS}
)

# Include directories
target_include_directories(implot3d PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}
)

# Link ImGui (ImPlot3D depends only on ImGui)
target_link_libraries(implot3d PUBLIC imgui)

# Set C++ standard
set_target_properties(implot3d PROPERTIES
    CXX_STANDARD 17
    CXX_STANDARD_REQUIRED ON
)
